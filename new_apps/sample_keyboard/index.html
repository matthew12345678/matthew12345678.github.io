    <script>
    const letters = ['Q','W','E','R','T','Y','U','I','O','P'];
    const keyContainer = document.getElementById("keys");
    let audioBuffers = {};
    let activeSources = {};
    let loopEnabled = true; // LOOP IS ALWAYS ON (hidden but active)
    const audioContext = new (window.AudioContext || window.webkitAudioContext)();
    const masterGain = audioContext.createGain();
    masterGain.connect(audioContext.destination);
    masterGain.gain.value = 1;
    // Unlock audio on first click
    function unlockAudio() {
    if (audioContext.state === 'suspended') {
    audioContext.resume();
    }
    }
    document.body.addEventListener("click", unlockAudio, { once: true });
    document.body.addEventListener("touchstart", unlockAudio, { once: true });
    // Create keys UI
    letters.forEach(letter => {
    const div = document.createElement("div");
    div.classList.add("key");
    div.innerText = letter;
    div.dataset.key = letter;
    keyContainer.appendChild(div);
    div.addEventListener("mousedown", (e) => {
    e.preventDefault();
    startSound(letter);
    });
    div.addEventListener("mouseup", () => stopSound(letter));
    div.addEventListener("mouseleave", () => stopSound(letter));
    // Touch support
    div.addEventListener("touchstart", (e) => {
    e.preventDefault();
    startSound(letter);
    });
    div.addEventListener("touchend", () => stopSound(letter));
    div.addEventListener("touchcancel", () => stopSound(letter));
    });
    // Load samples from folder
    document.getElementById("folderInput").addEventListener("change", async function(e) {
    const files = Array.from(e.target.files).filter(f =>
    f.type.includes("audio") || f.name.match(/\.(wav|mp3)$/i)
    );
    if (files.length === 0) {
    alert("No audio files found! Please select a folder with .wav or .mp3 files.");
    return;
    }
    let loadedCount = 0;
    for (let i = 0; i < letters.length && i < files.length; i++) {
    if (files[i]) {
    try {
    const arrayBuffer = await files[i].arrayBuffer();
    audioBuffers[letters[i]] = await audioContext.decodeAudioData(arrayBuffer);
    loadedCount++;
    } catch(err) {
    console.warn(`Failed to load ${files[i].name}:`, err);
    }
    }
    }
    if (loadedCount > 0) {
    alert(`Loaded ${loadedCount} sounds! Press Q W E R T Y U I O P keys to play. (Loop mode ON)`);
    } else {
    alert("Could not decode audio files. Please ensure valid .wav or .mp3 files.");
    }
    });
    function startSound(letter) {
    if (!audioBuffers[letter]) return;
    // Resume context if needed
    if (audioContext.state === 'suspended') {
    audioContext.resume();
    }
    // Stop existing instance if playing (retrigger behavior)
    if (activeSources[letter]) {
    try {
    activeSources[letter].stop();
    } catch(e) {}
    delete activeSources[letter];
    }
    const source = audioContext.createBufferSource();
    source.buffer = audioBuffers[letter];
    source.loop = loopEnabled; // Always true
    source.playbackRate.value = parseFloat(speedSlider.value);
    source.connect(masterGain);
    source.start();
    activeSources[letter] = source;
    // Visual feedback
    const keyElement = document.querySelector(`.key[data-key="${letter}"]`);
    if (keyElement) keyElement.classList.add("active");
    }
    function stopSound(letter) {
    if (activeSources[letter]) {
    try {
    activeSources[letter].stop();
    } catch(e) {}
    delete activeSources[letter];
    }
    const keyElement = document.querySelector(`.key[data-key="${letter}"]`);
    if (keyElement) keyElement.classList.remove("active");
    }
    // Keyboard controls
    document.addEventListener("keydown", (e) => {
    const letter = e.key.toUpperCase();
    if (letters.includes(letter)) {
    e.preventDefault();
    if (!activeSources[letter]) {
    startSound(letter);
    }
    }
    });
    document.addEventListener("keyup", (e) => {
    const letter = e.key.toUpperCase();
    if (letters.includes(letter)) {
    e.preventDefault();
    stopSound(letter);
    }
    });
    // Sliders
    const volumeSlider = document.getElementById("volume");
    const speedSlider = document.getElementById("speed");
    volumeSlider.addEventListener("input", (e) => {
    masterGain.gain.value = parseFloat(e.target.value);
    });
    speedSlider.addEventListener("input", (e) => {
    const newRate = parseFloat(e.target.value);
    // Update all currently playing sounds
    Object.values(activeSources).forEach(source => {
    if (source && source.playbackRate) {
    source.playbackRate.value = newRate;
    }
    });
    });
    // Loop is always ON - no toggle button visible
    const loopBtn = document.getElementById("loopBtn");
    if (loopBtn) {
    loopBtn.style.pointerEvents = "none";
    }
    // Cleanup any stray ended sources
    setInterval(() => {
    for (let [letter, source] of Object.entries(activeSources)) {
    if (!source || source.playbackState === 'ended') {
    delete activeSources[letter];
    const el = document.querySelector(`.key[data-key="${letter}"]`);
    if (el) el.classList.remove("active");
    }
    }
    }, 500);
    </script>
